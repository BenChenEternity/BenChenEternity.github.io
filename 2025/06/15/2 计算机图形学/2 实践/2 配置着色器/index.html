2 配置着色器

NicolasDavout 2025-06-15 19:02:00
Categories： Tags：

.vert 文件

// 指定 GLSL 版本为 3.30，并使用核心模式(Core-profile)
#version 330 core

// --- 输入变量 (Vertex Attributes) ---
// 从 VBO 接收顶点数据。layout(location = N) 将变量绑定到第 N 个顶点属性。

// layout(location = 0): 绑定到第0个属性，通常是位置数据
// in vec3: 输入一个三维向量
// aPos: 变量名，代表顶点位置 (Position)
layout (location = 0) in vec3 aPos;

// layout(location = 1): 绑定到第1个属性，通常是颜色数据
// aColor: 变量名，代表顶点颜色 (Color)
layout (location = 1) in vec3 aColor;


// --- 输出变量 ---
// 将数据传递给下一个着色器阶段（片元着色器）

// out vec3: 输出一个三维向量
// ourColor: 变量名，用于将颜色传递出去
out vec3 ourColor;


// --- 全局变量 (Uniforms) ---
// 从 CPU 程序传入的、对所有顶点都相同的全局变量

// mat4: 4x4 矩阵类型
// model: 模型矩阵 (将模型局部坐标 -> 世界坐标)
// view: 视图矩阵 (将世界坐标 -> 观察空间坐标)
// projection: 投影矩阵 (将观察空间坐标 -> 裁剪空间坐标)
uniform mat4 model;
uniform mat4 view;
uniform mat4 projection;


// --- 主函数 ---
// 每个顶点都会执行一次 main 函数
void main()
{
    // gl_Position 是一个内置的、必须被赋值的 vec4 输出变量
    // 它决定了顶点在屏幕上的最终位置
    // 计算过程：通过 MVP 矩阵链式相乘，将顶点从局部空间变换到裁剪空间
    // 注意：矩阵乘法顺序为 投影 * 视图 * 模型
    // vec4(aPos, 1.0) 是将三维位置坐标转换为四维齐次坐标，w分量为1.0
    gl_Position = projection * view * model * vec4(aPos, 1.0);

    // 将输入的顶点颜色直接赋值给输出变量
    // 这个颜色值会经过插值后，传递给片元着色器
    ourColor = aColor;
}

// 总结：
// 1. 主要任务：计算每个顶点的最终裁剪空间位置 (gl_Position)。
// 2. 附加任务：将顶点颜色 (aColor) 原封不动地传递给片元着色器 (ourColor)。
